Create docker container:
1. docker-compose build
2. docker-compose up -d

============================
Upload to Docker Hub
1. docker login
2. docker push lamvannguyen/digih2o-platform:v1   # tag

On a new computer:
1. Create a docker-compose.yml file:

services:
  backend:
    image: lamvannguyen/digih2o-platform:v1
    container_name: digi_H2O
    restart: always
    ports:
      - "8000:8000"
    env_file: .env
    volumes:
      - backend_static:/app/static
volumes:
  backend_static:

2. docker-compose pull
3. docker-compose up -d
